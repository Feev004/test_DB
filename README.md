# test_DB

# 1. สร้างโฟลเดอร์โปรเจคก่อน ด้วยมือนักศึกษา

# 2. สร้างไฟล์ package.json (ไฟล์สำหรับเก็บรายชื่อ Library ที่ใช้)

    npm init -y

# 3. ติดตั้ง Express.js

    npm i express
    npm i mysql2
    npm i cors
    npm install multer

# i ย่อมาจาก install

# 4. การรัน server.js

"scripts": {
    "start": "node server.js"
}
npm start

# อิงจาก "main": "server.js" ที่อยู่ใน package.json
